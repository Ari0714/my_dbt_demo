{% macro select_with_target_columns_regex(source_sql, target_relation) %}
{#
    è‡ªåŠ¨ç”Ÿæˆ SELECT åˆ—è¡¨ï¼ŒåŸºäºç›®æ ‡è¡¨å­—æ®µè¡¥é½ç¼ºåˆ—ã€‚
    - source_sql: æº SQL å†…å®¹ï¼ˆå­—ç¬¦ä¸²ï¼‰ï¼Œç”¨äºæ­£åˆ™æå–åˆ—å
    - target_relation: ç›®æ ‡è¡¨ï¼ˆthisï¼‰
#}

    {# ---------- Step 1. ä»æº SQL ä¸­æå–åˆ—å ---------- #}
    {% set clean_sql = source_sql
        | replace('\n', ' ')
        | replace('\r', ' ')
        | replace('\t', ' ')
        | regex_replace('\s+', ' ')
    %}

    {% set matches = clean_sql | regex_findall('(?is)select\\s+(.*?)\\s+from') %}

    {% if matches | length == 0 %}
        {{ exceptions.raise_compiler_error("æ— æ³•åœ¨æº SQL ä¸­æ‰¾åˆ° SELECT ... FROM ç»“æ„") }}
    {% endif %}

    {% set field_str = matches[0] %}
    {% set field_str = field_str | regex_replace('(?i)\\sas\\s', ' ') %}
    {% set raw_cols = field_str.split(',') %}
    {% set src_names = [] %}

    {% for c in raw_cols %}
        {% set col_name = c.strip() %}
        {% if ' ' in col_name %}
            {% set col_name = col_name.split(' ') | last %}
        {% endif %}
        {% do src_names.append(col_name | trim) %}
    {% endfor %}

    {# ---------- Step 2. è·å–ç›®æ ‡è¡¨å­—æ®µ ---------- #}
    {% set tgt_cols = adapter.get_columns_in_relation(target_relation) %}
    {% if tgt_cols is none %}
        {{ exceptions.raise_compiler_error("æ— æ³•è·å–ç›®æ ‡è¡¨åˆ—: " ~ target_relation) }}
    {% endif %}

    {% set tgt_names = [] %}
    {% for col in tgt_cols %}
        {% do tgt_names.append(col.name|string) %}
    {% endfor %}

    {# ---------- Step 3. æ„é€  SELECT åˆ—åˆ—è¡¨ ---------- #}
    {% set select_list = [] %}
    {% for col in tgt_names %}
        {% if col in src_names %}
            {% do select_list.append(col) %}
        {% else %}
            {% do select_list.append("NULL AS " ~ col) %}
        {% endif %}
    {% endfor %}

    {{ return(select_list | join(',\n  ')) }}
{% endmacro %}

    CREATE OR REPLACE TEMPORARY VIEW temp_product_stats AS (
        SELECT
            product_id,

ORDER BY s.total_revenue DESC
